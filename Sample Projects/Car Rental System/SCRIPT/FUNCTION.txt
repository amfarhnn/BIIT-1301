--7.2.1 Calculate 10% Discount for Payment that Exceed RM80 and 5% for Less Than RM80


SET SERVEROUTPUT ON
SET VERIFY OFF

CREATE OR REPLACE FUNCTION calDisc(totalPay in number) return number is 
  v_disc number ;

BEGIN
  if totalPay >= 80 then
    v_disc := 0.10;
  else
    v_disc := 0.05;
  end if;
  return v_disc;
END calDisc;

dbms_output.put_line('Final total payment is RM '||v_afterDisc);



END;




--	7.2.2 Add a 2% Service Charge for Online Payments and a 4% Service Charge for Offline Payments






SET SERVEROUTPUT ON;
SET VERIFY OFF;

CREATE OR REPLACE FUNCTION serviceCharge(p_clientid payment.clientid%type) RETURN NUMBER IS 

v_chg NUMBER(10,2);
v_total payment.totalpayment%type;
v_type payment.paymenttype%type;

BEGIN  
   SELECT paymenttype, totalpayment
   INTO v_type, v_total
   FROM payment
   WHERE clientid = p_clientid;
  
   IF v_type = 'ONLINE' THEN
    v_chg := 0.2;
    DBMS_OUTPUT.PUT_LINE('Service Charge: 2%');
   ELSE 
    v_chg := 0.4;
    DBMS_OUTPUT.PUT_LINE('Service Charge: 4%');
   END IF; 
   
   v_chg := v_chg * v_total + v_total;
   
  RETURN v_chg;

END;
