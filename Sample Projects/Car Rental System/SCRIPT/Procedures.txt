-- 7.1.1 Allow Admin to Update Return Date 

SET SERVEROUTPUT ON
SET VERIFY OFF

CREATE OR REPLACE PROCEDURE updateNewReturn 
(book_ID number,c_id number,newReturn date) IS


BEGIN 

update booking
SET returnDate = TO_DATE (newReturn)
where bookingID=book_ID
AND clientID=c_id;

DBMS_OUTPUT.PUT_LINE('DATA UPDATED ');

END;



--anonymous block 

ACCEPT booking_ID PROMPT 'Enter the bookingID:'
ACCEPT client_ID PROMPT 'Enter the clientID:'
ACCEPT newRDate  PROMPT 'Enter  new return date :'

DECLARE

v_oldDate date;
v_clientName client.clientName%type;
v_newDate date:= '&newRDate';

BEGIN

SELECT returnDate 
INTO v_oldDate 
FROM booking
WHERE bookingID='&booking_ID'
AND clientID='&client_ID';

SELECT clientName 
into v_clientName
FROM client
where clientID='&client_ID';

updateNewReturn ('&booking_ID','&client_ID','&newRDate');
DBMS_OUTPUT.PUT_LINE('The  return date for Mr./Mrs/Miss '|| v_clientName ||' are updated successfully');
DBMS_OUTPUT.PUT_LINE('The original return date was  '|| v_oldDate);
DBMS_OUTPUT.PUT_LINE('The new return date is  '|| v_newDate);
END;




--7.1.2 Allow Admin to View Client Profile

set serveroutput on
set verify off
 
create or replace procedure ClientDetails
(c_id client.clientid%type)
 
is
c_clientname client.clientname%type;
c_clientcontact client.clientcontact%type;
c_clientaddr client.clientaddr%type;
c_clientpswrd client.clientpswrd%type;
 
begin
select clientName, clientContact, clientAddr, clientPswrd
into c_clientname, c_clientcontact, c_clientaddr, c_clientpswrd
from client
where clientid = c_id;
 
dbms_output.put_line('Client Name: '||c_clientname);
dbms_output.put_line('Client Contact: '||c_clientcontact);
dbms_output.put_line('Client Address: '||c_clientaddr);
dbms_output.put_line('Client Password: '||c_clientpswrd);
 
end;
/
 
--anonymous block
 accept c_id prompt 'Enter client ID: '
 
begin
ClientDetails ('&c_id');

end;
/

--7.1.4 Allow Admin to Add Client


SET SERVEROUTPUT ON
SET VERIFY OFF

CREATE OR REPLACE PROCEDURE add_client 
(c_id client.clientid%TYPE,
 c_name client.clientname%TYPE,
 c_contact client.clientcontact%TYPE,
 c_address client.clientadr%TYPE,
 c_password client.clientpswrd%TYPE)
 
 AS 
 
 BEGIN
 
   INSERT INTO client (clientID, clientName, clientContact, clientAdr, clientPswrd) 
   VALUES (c_id, c_name, c_contact, c_address, c_password); 
    
END;
 
ACCEPT c_id PROMPT 'Please enter client ID: ' 
ACCEPT c_name PROMPT 'Please enter name: ' 
ACCEPT c_contact PROMPT 'Please enter contact number: ' 
ACCEPT c_address PROMPT 'Please enter the address: ' 
ACCEPT c_password PROMPT 'Please enter the password: ' 

BEGIN 

add_client('&c_id', '&c_name', '&c_contact', '&c_address', '&c_password');

DBMS_OUTPUT.PUT_LINE('The client name has been successfully added');
END;



--7.1.5 Show Available Car


SET SERVEROUTPUT ON
SET VERIFY OFF

CREATE OR REPLACE PROCEDURE listcar(p_carstatus VARCHAR2) IS 

CURSOR carDetails IS
 SELECT carid, carname
 FROM car
 WHERE carstatus = p_carstatus;

BEGIN 
 DBMS_OUTPUT.PUT_LINE ('List of available cars:');
 
 FOR book IN carDetails LOOP
 IF carDetails%NOTFOUND THEN
 EXIT;
 ELSE
   DBMS_OUTPUT.PUT_LINE(carDetails%ROWCOUNT || '. ' || book.carid || ' : ' || book.carname);
 END IF; 
 END LOOP;

END listcar;
/

execute listcar('AVAILABLE');
